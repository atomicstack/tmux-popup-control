# Charmbracelet v2 Migration Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Upgrade bubbletea, bubbles, and lipgloss from v0.x to v2.0, replacing `muesli/reflow` with `charmbracelet/x/ansi`.

**Architecture:** All three charmbracelet packages moved to `charm.land` module paths in a coordinated v2 release. The migration is mechanical — import paths, renamed types, changed function signatures. All packages must upgrade together since bubbles v2 depends on bubbletea v2 and lipgloss v2.

**Tech Stack:** Go, Bubble Tea v2, Bubbles v2, Lipgloss v2, charmbracelet/x/ansi

**Step 2: Fetch new modules**

```bash
GOMODCACHE=$(pwd)/.gomodcache GOPROXY=https://proxy.golang.org,direct go mod download charm.land/bubbletea/v2@v2.0.1 charm.land/bubbles/v2@v2.0.0 charm.land/lipgloss/v2@v2.0.0 github.com/charmbracelet/x/ansi@v0.8.0
```

```go
// OLD
func (m *Model) View() string {
	// ... body builds a string ...

}

// NEW
func (m *Model) View() tea.View {
	header := m.menuHeader()
	var content string
	switch m.mode {
	case ModePaneForm:
		if m.paneForm != nil {
			content = m.viewPaneFormWithHeader(header)
			return m.wrapView(content)
		}
	case ModeWindowForm:
		if m.windowForm != nil {
			content = m.viewWindowFormWithHeader(header)
			return m.wrapView(content)
		}
	case ModeSessionForm:
		if m.sessionForm != nil {
			content = m.viewSessionFormWithHeader(header)
			return m.wrapView(content)
		}
	}
	if m.hasSidePreview() {
		content = m.viewSideBySide(header)
	} else {
		content = m.viewVertical(header)
	}
	return m.wrapView(content)
}

func (m *Model) wrapView(content string) tea.View {
	v := tea.NewView(content)
	v.AltScreen = true
	v.MouseMode = tea.MouseCell
	return v
}
```

**Step 3: Replace `msg.Type` switch with `msg.String()` switch**

The existing `msg.String()` switch block (lines 35-93) already uses string matching — keep it as-is.

Replace the `msg.Type` switch block (lines 94-141):

```go
// OLD
switch msg.Type {
case tea.KeyBackspace, tea.KeyCtrlH:
	if m.removeFilterRune() {
		return true, m.ensurePreviewForLevel(current)
	}
	return false, nil
case tea.KeyRunes:
	if msg.Alt {
		return false, nil
	}
	if len(msg.Runes) == 0 {
		return false, nil
	}
	for _, r := range msg.Runes {
		if unicode.IsControl(r) {
			return false, nil
		}
		if unicode.IsSpace(r) {
			return false, nil
		}
	}
	if m.appendToFilter(string(msg.Runes)) {
		return true, m.ensurePreviewForLevel(current)
	}
	return false, nil
case tea.KeySpace:
	if m.appendToFilter(" ") {
		return true, m.ensurePreviewForLevel(current)
	}
	return false, nil
case tea.KeyLeft:
	before := current.FilterCursorPos()
	if !current.MoveFilterCursorRuneBackward() {
		return false, nil
	}
	m.noteFilterCursorChange(current, before)
	events.Filter.Cursor(current.ID, current.FilterCursor)
	return true, nil
case tea.KeyRight:
	before := current.FilterCursorPos()
	if !current.MoveFilterCursorRuneForward() {
		return false, nil
	}
	m.noteFilterCursorChange(current, before)
	events.Filter.Cursor(current.ID, current.FilterCursor)
	return true, nil
}

// NEW
switch msg.String() {
case "backspace":
	if m.removeFilterRune() {
		return true, m.ensurePreviewForLevel(current)
	}
	return false, nil
case "space":
	if m.appendToFilter(" ") {
		return true, m.ensurePreviewForLevel(current)
	}
	return false, nil
case "left":
	before := current.FilterCursorPos()
	if !current.MoveFilterCursorRuneBackward() {
		return false, nil
	}
	m.noteFilterCursorChange(current, before)
	events.Filter.Cursor(current.ID, current.FilterCursor)
	return true, nil
case "right":
	before := current.FilterCursorPos()
	if !current.MoveFilterCursorRuneForward() {
		return false, nil
	}
	m.noteFilterCursorChange(current, before)
	events.Filter.Cursor(current.ID, current.FilterCursor)
	return true, nil
default:
	// Regular text input (replaces KeyRunes handling).
	if msg.Mod.Contains(tea.ModAlt) {
		return false, nil
	}
	text := msg.Text
	if text == "" {
		return false, nil
	}
	for _, r := range text {
		if unicode.IsControl(r) {
			return false, nil
		}
		if unicode.IsSpace(r) {
			return false, nil
		}
	}
	if m.appendToFilter(text) {
		return true, m.ensurePreviewForLevel(current)
	}
	return false, nil
}
```

Merge the two switch blocks into a single `switch msg.String()`. The first switch block (lines 35-93 with ctrl+u, ctrl+w, etc.) already uses `msg.String()`, so those cases just need to be combined into one unified switch. The `backspace`, `space`, `left`, `right` cases fold into it.
